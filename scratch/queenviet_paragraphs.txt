        // ============================================================
        // 4. TÍNH NĂNG: LĂN CHUỘT ĐỂ LẬT TRANG (KHÔNG TRÔI WEB)
        // ============================================================
        var container = document.getElementById('flipbookContainer');
        var isFlipping = false;

        // Sử dụng Javascript thuần để can thiệp sâu vào sự kiện cuộn (passive: false)
        container.addEventListener('wheel', function(e) {
            // Chặn trình duyệt cuộn trang
            e.preventDefault();
            e.stopPropagation();

            if(isFlipping) return;

            var delta = e.deltaY;
            var $container = $(container);
            var $nextBtn = $container.find('.df-ui-next');
            var $prevBtn = $container.find('.df-ui-prev');

            // Lăn xuống -> Next
            if (delta > 0) {
                if($nextBtn.length && !$nextBtn.hasClass('disabled')) {
                    $nextBtn.trigger('click');
                }
            } 
            // Lăn lên -> Prev
            else {
                if($prevBtn.length && !$prevBtn.hasClass('disabled')) {
                    $prevBtn.trigger('click');
                }
            }

            // Debounce: Chặn lật liên tục trong 0.5s
            isFlipping = true;
            setTimeout(function(){
                isFlipping = false;
            }, 500);

        }, { passive: false }); 
        // ============================================================
    }
});
